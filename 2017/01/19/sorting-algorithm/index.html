冒泡排序

基本思想：扫描整个数组，依次比较两个相邻数字的大小，如果不在正确的位置上，则交换位置。

function bubbleSort(array) {
    var length = array.length;
    for(var i = 0;i<length; i++){
        for(var j = i; j<length; j++){
            if(array[i]>array[j]){
                // 交换位置
        {array[i], array[j]} = {array[j], array[i]}
            }
        }
    }
    return array;
}

快速排序

插入排序

基本思想：从第一个元素开始，该元素可以认为已经被排序；取出下一个元素，在已经排序的元素序列中从后向前扫描；如果该元素（已排序）大于新元素，将该元素移到下一位置；重复步骤 3，直到找到已排序的元素小于或者等于新元素的位置；将新元素插入到该位置后；重复步骤 2~5。

function insertionSort(array) {
  for (var i = 1; i < array.length; i++) {
    var key = array[i]
    var j = i - 1
    while (j >= 0 && array[j] > key) {
      array[j + 1] = array[j]
      j--
    }
    array[j + 1] = key
  }
  return array
}

二分插入排序

基本思想：与插入排序的区别是在查找插入位置时使用二分查找，所以在速度上较插入排序有一定的提升。

function binaryInsertionSort(array) {
  for (var i = 1; i < array.length; i++) {
    var key = array[i],
      left = 0,
      right = i - 1
    while (left <= right) {
      var middle = parseInt((left + right) / 2)
      if (key > array[middle]) {
        left = middle + 1
      } else {
        right = middle - 1
      }
    }

    for (var j = i - 1; j >= left; j--) {
      array[j + 1] = array[j]
    }
    array[left] = key
  }
  return array
}
